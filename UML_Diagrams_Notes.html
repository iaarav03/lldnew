ğŸ’» C++ Implementation of IS-A

// Base class (Parent)
class Animal {
protected:
    string name;
    int age;
    
public:
    Animal(const string& n, int a) : name(n), age(a) {}
    
    // Virtual functions for polymorphism
    virtual void eat() {
        cout << name << " is eating." << endl;
    }
    
    virtual void sleep() {
        cout << name << " is sleeping." << endl;
    }
    
    virtual void makeSound() = 0; // Pure virtual (abstract)
    
    virtual ~Animal() = default;
};

// Derived class (Child) - Dog IS-A Animal
class Dog : public Animal {
private:
    string breed;
    
public:
    Dog(const string& n, int a, const string& b) 
        : Animal(n, a), breed(b) {}
    
    // Override parent methods
    void makeSound() override {
        cout << name << " says: Woof! Woof!" << endl;
    }
    
    // Dog-specific methods
    void bark() {
        cout << name << " is barking loudly!" << endl;
    }
    
    void wagTail() {
        cout << name << " is wagging tail happily!" << endl;
    }
    
    string getBreed() const { return breed; }
};

// Another derived class
class Cat : public Animal {
public:
    Cat(const string& n, int a) : Animal(n, a) {}
    
    void makeSound() override {
        cout << name << " says: Meow! Meow!" << endl;
    }
    
    void purr() {
        cout << name << " is purring contentedly." << endl;
    }
};

// Usage demonstrating polymorphism
int main() {
    // Creating objects
    Dog myDog("Buddy", 3, "Golden Retriever");
    Cat myCat("Whiskers", 2);
    
    // Using IS-A relationship - Dog IS-A Animal
    Animal* animals[] = {&myDog, &myCat};
    
    cout << "=== Polymorphic behavior ===" << endl;
    for (Animal* animal : animals) {
        animal->eat();        // Inherited from Animal
        animal->makeSound();  // Polymorphic call
        animal->sleep();      // Inherited from Animal
        cout << endl;
    }
    
    cout << "=== Dog-specific behavior ===" << endl;
    myDog.bark();      // Dog-specific method
    myDog.wagTail();   // Dog-specific method
    cout << "Breed: " << myDog.getBreed() << endl;
    
    return 0;
}

ğŸ¤” What is HAS-A?

HAS-A represents ownership or containment. One class "has" or "contains" another class as a member. There are two types:

  • ğŸ’ª Composition (Strong HAS-A): Owner controls lifetime
  • ğŸ¤ Aggregation (Weak HAS-A): Owner uses but doesn't control lifetime

ğŸ”§ Composition Characteristics

  • Ownership: Parent owns and controls child's lifecycle
  • Lifetime: Child cannot exist without parent
  • Example: Car HAS-A Engine, House HAS-A Room
  • UML Symbol: Filled diamond (â™¦)

ğŸ’» C++ Implementation of Composition

// Child class - Engine
class Engine {
private:
    int horsepower;
    string fuelType;
    bool isRunning;
    
public:
    Engine(int hp, const string& fuel) 
        : horsepower(hp), fuelType(fuel), isRunning(false) {
        cout << "Engine created with " << hp << " HP (" << fuel << ")" << endl;
    }
    
    ~Engine() {
        cout << "Engine destroyed" << endl;
    }
    
    void start() {
        if (!isRunning) {
            isRunning = true;
            cout << "Engine started! Vroom vroom!" << endl;
        }
    }
    
    void stop() {
        if (isRunning) {
            isRunning = false;
            cout << "Engine stopped." << endl;
        }
    }
    
    bool getRunningStatus() const { return isRunning; }
    int getHorsepower() const { return horsepower; }
};

// Parent class - Car (HAS-A Engine)
class Car {
private:
    string brand;
    string model;
    Engine engine;  // Composition: Car HAS-A Engine
    
public:
    // Engine is created when Car is created (composition)
    Car(const string& b, const string& m, int hp, const string& fuel)
        : brand(b), model(m), engine(hp, fuel) {  // Engine constructed here
        cout << "Car created: " << brand << " " << model << endl;
    }
    
    // Engine is automatically destroyed when Car is destroyed
    ~Car() {
        cout << "Car destroyed: " << brand << " " << model << endl;
    }
    
    void startCar() {
        cout << "Starting " << brand << " " << model << "..." << endl;
        engine.start();  // Delegating to engine
        if (engine.getRunningStatus()) {
            cout << "Car is ready to drive!" << endl;
        }
    }
    
    void stopCar() {
        cout << "Stopping " << brand << " " << model << "..." << endl;
        engine.stop();   // Delegating to engine
        cout << "Car stopped." << endl;
    }
    
    void displayInfo() const {
        cout << "Car: " << brand << " " << model 
             << " (Engine: " << engine.getHorsepower() << " HP)" << endl;
    }
};

int main() {
    cout << "=== Composition Example ===" << endl;
    
    {  // Scope to demonstrate lifecycle
        Car myCar("Toyota", "Camry", 200, "Gasoline");
        myCar.displayInfo();
        
        myCar.startCar();
        cout << "Driving around..." << endl;
        myCar.stopCar();
        
        cout << "End of scope - Car will be destroyed..." << endl;
    } // Car destructor called -> Engine destructor called automatically
    
    cout << "After scope - both Car and Engine are destroyed" << endl;
    return 0;
}

// Independent class - Student
class Student {
private:
    string name;
    int id;
    string major;
    
public:
    Student(const string& n, int studentId, const string& m) 
        : name(n), id(studentId), major(m) {
        cout << "Student created: " << name << " (ID: " << id << ")" << endl;
    }
    
    ~Student() {
        cout << "Student destroyed: " << name << endl;
    }
    
    void study() {
        cout << name << " is studying " << major << endl;
    }
    
    void takeExam() {
        cout << name << " is taking an exam" << endl;
    }
    
    string getName() const { return name; }
    int getId() const { return id; }
    string getMajor() const { return major; }
};

// Class that aggregates students
class University {
private:
    string name;
    string location;
    vector students;  // Aggregation: pointers to existing students
    
public:
    University(const string& n, const string& loc) 
        : name(n), location(loc) {
        cout << "University created: " << name << endl;
    }
    
    ~University() {
        cout << "University destroyed: " << name << endl;
        // Note: We don't delete students - they exist independently!
        cout << "Students continue to exist after university is destroyed" << endl;
    }
    
    // Enroll an existing student (aggregation)
    void enrollStudent(Student* student) {
        if (student) {
            students.push_back(student);
            cout << "Student " << student->getName() 
                 << " enrolled at " << name << endl;
        }
    }
    
    void removeStudent(int studentId) {
        auto it = remove_if(students.begin(), students.end(),
            [studentId](Student* s) { return s->getId() == studentId; });
        
        if (it != students.end()) {
            cout << "Student with ID " << studentId 
                 << " removed from " << name << endl;
            students.erase(it, students.end());
        }
    }
    
    void conductClass() {
        cout << "\n=== Class in session at " << name << " ===" << endl;
        for (Student* student : students) {
            student->study();
        }
    }
    
    void displayInfo() const {
        cout << "\n" << name << " (" << location << ")" << endl;
        cout << "Enrolled students: " << students.size() << endl;
        for (const Student* student : students) {
            cout << "  - " << student->getName() 
                 << " (ID: " << student->getId() 
                 << ", Major: " << student->getMajor() << ")" << endl;
        }
    }
};

int main() {
    cout << "=== Aggregation Example ===" << endl;
    
    // Create students independently
    Student* alice = new Student("Alice Johnson", 1001, "Computer Science");
    Student* bob = new Student("Bob Smith", 1002, "Mathematics");
    Student* charlie = new Student("Charlie Brown", 1003, "Physics");
    
    {  // University scope
        University mit("MIT", "Cambridge, MA");
        
        // Enroll existing students (aggregation)
        mit.enrollStudent(alice);
        mit.enrollStudent(bob);
        mit.enrollStudent(charlie);
        
        mit.displayInfo();
        mit.conductClass();
        
        // Remove a student (but student still exists)
        mit.removeStudent(1002);
        mit.displayInfo();
        
        cout << "\nUniversity scope ending..." << endl;
    } // University destroyed, but students still exist!
    
    cout << "\nAfter university destruction:" << endl;
    alice->study();    // Students still exist and work!
    charlie->study();
    
    // Manual cleanup (in real app, use smart pointers)
    delete alice;
    delete bob;
    delete charlie;
    
    return 0;
}

// Base class
class Vehicle {
protected:
    string id, brand;
public:
    virtual void start() = 0;
    virtual ~Vehicle() = default;
};

// IS-A relationship
class Car : public Vehicle {
private:
    int doors;
public:
    void start() override { cout << "Car engine started" << endl; }
    void openDoor() { cout << "Door opened" << endl; }
};

// HAS-A (Composition) - Rental owns vehicle reference
class Rental {
private:
    Vehicle* vehicle;  // Strong association
    Customer* customer;
    Date startDate, endDate;
public:
    Rental(Vehicle* v, Customer* c) : vehicle(v), customer(c) {}
    // Rental controls the rental relationship
};

// HAS-A (Aggregation) - Company uses vehicles
class RentalCompany {
private:
    vector fleet;  // Company has vehicles but doesn't own their lifecycle
    vector activeRentals;  // Company manages rentals (composition)
public:
    void addVehicle(Vehicle* v) { fleet.push_back(v); }
    Rental* createRental(Vehicle* v, Customer* c) {
        return new Rental(v, c);  // Company creates rental
    }
};
